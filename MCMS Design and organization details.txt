System Overview and Objectives 

This is a comprehensive design for a web application for the management of specialized medical centers, covering all required departments and ensuring the workflow from the moment the patient enters to discharges. The goal is to build a secure, scalable, multilingual system (Arabic/English), with a web frontend, a backend in Python, a powerful database, as well as accurate user permissions, detailed functional tests, and global drug data linkage. 

 

Proposed Architecture and Technologies 

Main Classes 

Web Frontend: React or Vue with Next.js/Nuxt for hybrid rendering, responsive design, medical  UI  components, and RTL support for Arabic. 

 Backend (Python): Django + DRF or FastAPI for REST/GraphQL design, user management, RBACs, logs, external integrations. 

Database: PostgreSQL for relationships and transactions, Redis for caching and queues, S3-compatible for rays and reports. 

Automated Messaging and Call-Up: WebSocket/Server-Sent Events for instant updates, Message Queue (RabbitMQ/Redis Streams) for events and calls. 

Medical integrations: FHIR/HL7 for clinical data capture, RxNorm/WHO ATC/DrugBank/UNICOM for drug standardization. 

Print & Reports:  HTML  + CSS  template engine for printing, with wkhtmltopdf or WeasyPrint to generate PDFs. 

Deployment and monitoring: Docker/Kubernetes, CI/CD, Prometheus + Grafana monitoring, centralized logs.

Django + DRF: Suitable if you want an integrated system with a robust  ORM, built-in user management, and high security.

# encounters/permissions.py 

from rest_framework.permissions import BasePermission 

 

class IsAdmin(BasePermission): 

    """ 

    Validity allows only users with the Admin role 

    """ 

    def has_permission(self, request, view): 

        return request.user.is_authenticated and request.user.role == 'admin' 

 

class IsProviderOrAdmin(BasePermission): 

    """ 

    Permission to allow service providers (Provider or Admin) 

    """

from reportlab.lib.pagesizes import A4 

from reportlab.pdfgen import canvas 

from django.conf import settings 

import os 

 

def generate_radiology_pdf(result): 

    """ 

    Generate a PDF radiology report with image and text report 

    """ 

    file_path = os.path.join(settings. MEDIA_ROOT, f"radiology/reports/result_{result.id}.pdf") 

    c = canvas. Canvas(file_path, pagesize=A4) 

 

    # Report Title 

    c.setFont("Helvetica-Bold", 16) 

    c.drawString(100, 800, "Radiology Report") 

 

    # Report Data 

    c.setFont("Helvetica", 12) 

    c.drawString(100, 770, f"Exam ID: {result.exam.id}") 

    c.drawString(100, 750, f"Report Date: {result.result_date.strftime('%Y-%m-%d %H:%M')}") 

    c.drawString(100, 730, f"Report: {result.report}") 

 

    # Insert the image if it exists 

    if result.image: 

        try: 

            c.drawImage(result.image.path, 100, 500, width=300, height=200) 

        except Exception: 

            c.drawString(100, 700, "Image could not be loaded.") 

 

    c.save() 

    return file_path

# appointments/models.py 

from django.db import models 

from django.utils import timezone 

import uuid 

 

class Appointment(models. Model): 

    STATUS_CHOICES = [ 

        ('pending', 'Pending'), 

        ('confirmed', 'Confirmed'), 

        ('cancelled', 'Cancelled'), 

    ] 

 

    patient = models. ForeignKey("patients. Patient", on_delete=models. CASCADE) 

    scheduled_at = models. DateTimeField() 

    status = models. CharField(max_length=20, choices=STATUS_CHOICES, default='pending') 

    confirmation_token = models. UUIDField(default=uuid.uuid4, editable=False) 

    confirmation_deadline = models. DateTimeField(default=lambda: timezone.now() + timezone.timedelta(hours=24)) 

 

    def is_expired(self): 

        return timezone.now() > self.confirmation_deadline 

 

📡 APIs (Serializers + Views) 

# appointments/serializers.py 

from rest_framework import serializers 

from .models import Appointment 

 

class AppointmentSerializer(serializers. ModelSerializer): 

    class Meta: 

        model = Appointment 

        fields = ['id', 'patient', 'scheduled_at', 'status', 'confirmation_deadline'] 

 

class AppointmentConfirmSerializer(serializers. Serializer): 

    token = serializers. UUIDField() 

# appointments/views.py 

from rest_framework import status, generics 

from rest_framework.response import Response 

from .models import Appointment 

from .serializers import AppointmentSerializer, AppointmentConfirmSerializer 

 

class AppointmentConfirmView(generics. GenericAPIView): 

    serializer_class = AppointmentConfirmSerializer 

 

    def post(self, request, *args, **kwargs): 

        serializer = self.get_serializer(data=request.data) 

        serializer.is_valid(raise_exception=True) 

        token = serializer.validated_data['token'] 

 

        try: 

            appointment = Appointment.objects.get(confirmation_token=token) 

        except Appointment.DoesNotExist: 

            return Response({"error": "Invalid token"}, status=status. HTTP_404_NOT_FOUND) 

 

        if appointment.is_expired(): 

            appointment.status = 'cancelled' 

            appointment.save() 

            return Response({"error": "Confirmation deadline expired"}, status=status. HTTP_400_BAD_REQUEST) 

 

        appointment.status = 'confirmed' 

        appointment.save() 

        return Response(AppointmentSerializer(appointment).data, status=status. HTTP_200_OK)
